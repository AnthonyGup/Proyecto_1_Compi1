$ Preguntas dentro de IF y FOR usando SECTION
number mostrar_pregunta = 1
IF (mostrar_pregunta == 1)
{
    SECTION [
        width: 420,
        height: 180,
        pointX: 540,
        pointY: 280,
        elements: {
            OPEN_QUESTION [
                label: "Deseas continuar?"
            ]
        }
    ]
}

FOR (i in 1 .. 2)
{
    SECTION [
        width: 420,
        height: 180,
        pointX: 540,
        pointY: 470,
        elements: {
            DROP_QUESTION [
                label: "Pregunta iterada " + i,
                options: {"A", "B"}
            ]
        }
    ]
}
